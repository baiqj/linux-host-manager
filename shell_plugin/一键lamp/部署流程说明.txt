centos 6.3

yum -y install httpd httpd-tools

yum -y install mysql mod_auth_mysql mysql-connector-odbc mysql-server 

3、安装php
yum -y install php php-mysql php-mbstring php-imap php-gd php-fpm php-intl php-xml php-xmlrpc php-devel
